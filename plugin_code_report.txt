#### 2.2.4 AICartographer.cpp
```cpp
#include "AICartographer.h"
#include "Widgets/Docking/SDockTab.h"
#include "SWebBrowser.h"
#include "WorkspaceMenuStructure.h"
#include "WorkspaceMenuStructureModule.h"

static const FName AICartographerTabName("AICartographer");

void FAICartographerModule::StartupModule()
{
    FGlobalTabmanager::Get()->RegisterNomadTabSpawner(AICartographerTabName, FOnSpawnTab::CreateRaw(this, &FAICartographerModule::OnSpawnPluginTab))
        .SetDisplayName(FText::FromString(TEXT("AICartographer Web UI")))
        .SetMenuType(ETabSpawnerMenuType::Enabled);
}

void FAICartographerModule::ShutdownModule()
{
    FGlobalTabmanager::Get()->UnregisterNomadTabSpawner(AICartographerTabName);
}

TSharedRef<SDockTab> FAICartographerModule::OnSpawnPluginTab(const FSpawnTabArgs& SpawnTabArgs)
{
    return SNew(SDockTab)
        .TabRole(ETabRole::NomadTab)
        [
            SNew(SWebBrowser)
            .InitialURL("http://localhost:5173/")
            .ShowControls(false)
            .SupportsTransparency(true)
        ];
}

IMPLEMENT_MODULE(FAICartographerModule, AICartographer)
```
**Analysis**: Complete implementation of NomadTab registration. On startup, registers a dockable tab that embeds SWebBrowser pointing to local React dev server (port 5173). Properly unregisters the tab on module shutdown.
